	\begin{UseCase}{CU A12}{Gestionar Proyectos Pre-registrados}{Administra un catálogo en el que se puede consultar, aceptar o rechazar el Pre-registro de un Proyecto}
		\UCitem{Versión}{1.3}
		\UCitem{Actor(es)}{Administrador.}
		\UCitem{Propósito}{Visualizar los Proyectos Pre-registrados de una forma ordenada.}
		\UCitem{Resumen}{Se muestran los Proyectos Pre-registrados por Usuarios no registrados con la posibilidad de evaluar el Proyecto.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Despliegue de los Proyectos Pre-registrados.}
		\UCitem{Precondiciones}{Ninguna.}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Autor}{Pérez Domínguez Alberto Michel}
		\UCitem{Referencias}{SIDAM-BESP-P1-Especificacion de Catálogos}
		\UCitem{Tipo}{Primario.}
		\UCitem{Módulo}{Administración}
	\end{UseCase}
	
	\begin{UCtrayectoria}{Principal}
		\UCpaso[\UCactor] Oprime el botón \IUbutton{Gestionar Proyectos Pre-registrados} en el menú \IUref{IUMenuAdministrador}{Menú del Administrador}.
		\UCpaso Busca los Proyectos Pre-registrados por Usuarios no registrados. \label{paso:CUA12BuscarProyectos}
		\UCpaso Muestra la pantalla \IUref{IUGestProyectos}{Gestionar Proyectos Preregistrados} donde se despliegan los de datos de los Proyectos, con la posibilidad de Evaluar el Pre-registro.\UCExtensionPoint{CU A12.1}{Evaluar Pre-registro de Proyecto}
	\end{UCtrayectoria}

%--------- Evaluar Proyecto
	\begin{UseCase}{CU A12.1}{Evaluar Pre-registro de Proyecto}{El Administrador evalua el Pre-registro de un Proyecto.}
			\UCitem{Versión}{1.3}
			\UCitem{Actor(es)}{Administrador.}
			\UCitem{Propósito}{Consultar los datos del proyecto, aceptar o rechazar el pre-registro.}
			\UCitem{Resumen}{Se consultan los datos de un Proyecto y su responsable con la posibilidad de Aceptar o Rechazar el Proyecto.}
			\UCitem{Entradas}{Identificador del Proyecto seleccionado.}
			\UCitem{Salidas}{Datos del Proyecto seleccionado\ref{dd:ProyectoPreregistrado} y datos del Responsable\ref{datosResponsable}.}
			\UCitem{Precondiciones}{Debe existir el Pre-registro del Proyecto seleccionado.}
			\UCitem{Postcondiciones}{Ninguna.}
			\UCitem{Autor}{Pérez Domínguez Alberto Michel}
			\UCitem{Referencias}{SIDAM-BESP-P1-Especificacion de Catálogos}
			\UCitem{Tipo}{Secundario. Viene del \UCref{CU A12}}
			\UCitem{Módulo}{Administración}
	\end{UseCase}

	\begin{UCtrayectoria}{Principal}
			\UCpaso[\UCactor] Oprime el botón \IUbutton{Evaluar} correspondiente al Pre-registro del Proyecto que se encuentra en la Tabla correspondiente a ``Proyectos Pre-registrados por Usuarios no registrados'' de la pantalla \IUref{IUGestProyectos}{Gestionar Proyectos Preregistrados}.
			\UCpaso Muestra la pantalla \IUref{IUCrearCuenta}{Evaluar Pre-registro de Proyecto y Consolidar Usuario} con los datos del Proyecto seleccionado, la opción \textit{``Crear Cuenta de Usuario''} \textbf{activada}, los \textit{``Datos de Pre-registro''} cargados en las casillas de \textit{``Responsable''}, la opción \textit{``Aceptado''} y el botón \IUbutton{Enviar} permanecen bloqueados.\label{CrearCuenta}
			\UCpaso[\UCactor] Oprime el botón \IUbutton{Agregar}.\Trayref{A}
			\UCpaso Lleva a cabo la trayectoria del caso de uso CU A1.1 Agregar Usuario \UCref{CU A1.1} con los datos del pre-registro .\UCExtensionPoint{CU A1.1}{Agregar Usuario}
			\UCpaso Una vez cumplida la trayectoria normal del caso de uso \textbf{Agregar Usuario} se desbloquea la opción \textit{``Aceptado''} y el botón \IUbutton{Enviar}.
			\UCpaso [\UCactor] Verifica los datos del Proyecto.\label{paso:CUA12.1VerifcaDatos}
			\UCpaso [\UCactor] Selecciona la opción ``Aceptado'' e ingresa Observaciones.\label{paso:CUA12.1Aceptar}\Trayref{B}
			\UCpaso [\UCactor] Oprime el botón \IUbutton{Enviar}.\Trayref{C}
			\UCpaso Cambia el Estado del Proyecto de Pre-registrado a Aceptado (o en Edición)\ref{IUEstadoProyecto}.
			\UCpaso Guarda las observaciones y las Envía al correo electrónico principal del Usuario.\label{paso:CUA12.1EnviarObservacion}
			\UCpaso Muestra el mensaje (MSG-4) de operación exitosa.\ref{MSG4}
			\UCpaso Continúa en el paso \ref{paso:CUA12BuscarProyectos} del \UCref{CU A12}.
	\end{UCtrayectoria}

	\begin{UCtrayectoriaA}{A}{Reasignación de Responsable}{El Administrador reasigna el Responsable del Proyecto a un Usuario ya registrado.}
			\UCpaso[\UCactor] Selecciona la opción \textit{``Utilizar cuenta de Usuario existente''}.
			\UCpaso Muestra en la pantalla la lista de Usuarios registrados\IUref{IUUsarCuenta}{Evaluar Proyecto y usar Cuenta de Usuario existente}.
			\UCpaso [\UCactor] Ingresa el nombre del Usuario en el campo Buscar.
			\UCpaso Muestra los registros que coinciden con los datos de busqueda.\Trayref{A.1}
			\UCpaso [\UCactor] Selecciona el registro del Usuario.
			\UCpaso Asigna el Usuario ya registrado al Pre-registro del Proyecto.
			\UCpaso Continúa en el paso \ref{paso:CUA12.1VerifcaDatos} del \UCref{CU A12.1}.
	\end{UCtrayectoriaA}

	\begin{UCtrayectoriaA}{A.1}{Registro de Usuario no encontrado}{Los datos ingresados en el campo de busqueda no coinciden con los datos de los registros de Usuarios.}
			\UCpaso Muestra al Administrador el mensaje (MSG-5) indicando que no se encontrarón resultados.\ref{MSG-5}
			\UCpaso[\UCactor] Selecciona la opción \textit{``Crear Cuenta de Usuario''}.
			\UCpaso Continua en el paso \ref{CrearCuenta} de la \Trayref{A}.
	\end{UCtrayectoriaA}

	\begin{UCtrayectoriaA}{B}{Cancela Proyecto}{El Administrador cancela el Proyecto.}
			\UCpaso[\UCactor] Decide no Aceptar el Pre-registro del Proyecto.
			\UCpaso[\UCactor] Selecciona la opción ``Rechazado'' e ingresa Observaciones.
			\UCpaso Cambia el Estado del Proyecto de Pre-registro a Cancelado.
			\UCpaso Continúa en el paso \ref{paso:CUA12.1EnviarObservacion} del \UCref{CU A12.1} 
	\end{UCtrayectoriaA}

	\begin{UCtrayectoriaA}{C}{Cancelar operación}{El Administrador abandona el Caso de Uso.}
			\UCpaso[\UCactor] Decide no realizar la evaluación del Pre-registro del Proyecto.
			\UCpaso[\UCactor] Oprime el botón \IUbutton{Regresar}.
			\UCpaso Continúa en el paso \ref{paso:CUA12BuscarProyectos} del \UCref{CU A12} 
	\end{UCtrayectoriaA}
