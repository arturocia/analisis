%-------------------------------------- CUDG1 Consultar proyectos

  \begin{UseCase}{CUDG1}{Consultar proyectos}{Permite acceder a una lista con los proyectos actuales.}
    \UCitem{Versión}{1.0}
    \UCitem{Actor(es)}{Director general.}
    \UCitem{Propósito}{Conocer y consultar los proyectos que se encuentran actualmente en el sistema.}
    \UCitem{Resumen}{Enlista los proyectos para su consulta.}
    \UCitem{Entradas}{Ninguna.}
    \UCitem{Salidas}{Lista con los proyectos.}
    \UCitem{Precondiciones}{Al menos debe haber un proyecto registrado.}
    \UCitem{Postcondiciones}{Ninguna.}
    \UCitem{Autor}{Pedro Hernandez.}
    \UCitem{Referencias}{SIDAM-BESP-P1}
    \UCitem{Tipo}{Primario.}
    \UCitem{Módulo}{Dirección general.}
  \end{UseCase}
  
  \begin{UCtrayectoria}{Principal}
    \UCpaso[\UCactor] Selecciona la opción \IUbutton{Consultar proyecto} de la pantalla \IUref{IUMenuDirector}{Menú de Dirección General}.
    \UCpaso Busca los Proyectos actuales.\Trayref{A}
    \UCpaso Muestra la pantalla \IUref{IUConsultarProyecto}{Consultar proyecto} donde se despliegan los proyectos y la opción de Revisar el proyecto.
    \UCExtensionPoint{CU2.1}{Revisión de Proyecto.}
  \end{UCtrayectoria}

  \begin{UCtrayectoriaA}{A}{No hay proyectos registrados}{No se encontraron registros.}
    \UCpaso Muestra mensaje (MSG-5) indicando que no se encontraron Proyectos.\ref{MSG5}
  \end{UCtrayectoriaA}
