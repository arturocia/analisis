%-------------------------------------- CU2.1 Revisión de proyectos.


	\begin{UseCase}{CU2.1}{Revisión de Proyecto}{Visualizar y/o gestionar la definición de un proyecto según el perfil del actor.}
		\UCitem{Versión}{1.0}

		\UCitem{Actor(es)}{Coordinador, Secretario, Gerente.}
		\UCitem{Propósito}{Consultar la definición de un proyecto.}
		\UCitem{Resumen}{Agrupa las funcionalidades de consulta de información relacionada con un proyecto.}
		\UCitem{Entradas}{Ninguna.}
		\UCitem{Salidas}{Ninguna.}
		\UCitem{Precondiciones}{
			Para el coordinador:
			\begin{itemize}
 				\item Haber pasado por el \UCref{CU0} y obtenido perfil de Coordinador.
 				\item Que exista un proyecto cuyo responsable sea el Coordinador.
			\end{itemize}
			Para los demas usuarios:
			\begin{itemize}
				\item Haber pasado por el \UCref{CU2}.
			\end{itemize}
		}
		\UCitem{Postcondiciones}{Ninguna.}
		\UCitem{Autor}{Jessica Ramos. }
		\UCitem{Referencias}{}
		\UCitem{Tipo}{Secundario. Viene del \UCref{CU2}.}Coordinación
		\UCitem{Módulo}{Coordinación, Secretaría y Gerencia.}
	\end{UseCase}
	
	\begin{UCtrayectoria}{Principal}

		\UCpaso[\UCactor] Selecciona el proyecto dependiendo del usuario:
		\begin{itemize}
		    \item Para el Coordinador selecciona el proyecto del menú que se muestra de lado izquierdo en pantalla. \IUref{IUMenuCoordinador}{Menú del Coordinador}.
		    \item Para el Gerente y Secretaria selecciona el proyecto del diagrama de Gantt en la pantalla \IUref{IURevisarPrograma}{Revisar Programas}.
		\end{itemize}
		\UCpaso Busca datos del proyecto seleccionado.
		\UCpaso Muestra los datos en pantalla \IUref{IURevisarProyectos}{Revisión de Proyectos} donde se despliegan los datos generales del proyecto y las opciones para su gestión según el perfil del usuario. \UCExtensionPoint{CU2.1.1}{Revisar información de un proyecto}, \UCExtensionPoint{CUC2}{Planeación}, \UCExtensionPoint{CU2.1.3}{Revisar seguimiento del proyecto}, \UCExtensionPoint{CU2.1.2}{Revisar avances de un proyecto}, \UCExtensionPoint{CUG1}{Revisión de proyectos}

	\end{UCtrayectoria}
